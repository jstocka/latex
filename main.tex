% Komenda dla pierwszego rozdziału (bez numeracji)
\newcommand{\firstchapter}[1]{%
  \titleformat{\chapter}[display]
    {\normalfont\fontsize{14}{16.8}\bfseries\centering} % 14pt, pogrubione, wyśrodkowane
    {}
    {0pt}
    {\normalfont\fontsize{14}{16.8}\bfseries\centering}
  \chapter{#1}
  \vspace{\chapterspace}
  % Przywracanie oryginalnego formatu dla kolejnych rozdziałów
  \titleformat{\chapter}[display]
    {\normalfont\fontsize{14}{16.8}\bfseries\centering} % 14pt, pogrubione, wyśrodkowane
    {\chaptertitlename~\thechapter} % "Chapter X" w jednej linii
    {0.1em} % odstęp między "Chapter X" a tytułem
    {\normalfont\fontsize{14}{16.8}\bfseries\centering}
}
